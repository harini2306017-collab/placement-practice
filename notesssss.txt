#include <stdlib.h>

int main()
{
    int n;
    scanf("%d",&n); //145
    int temp=n;
    int fact,sum=0;
    while(n!=0)  //145!=0  14!=0  1!=0
    {
        int dig=n%10; //dig=5 4
        fact=1;
        for (int itr=1;itr<=dig;itr+=1)  // 1 2 3 4 5  5    1 2 3 4  4   1  1
            fact*=itr;  //120  24 1
        n/=10;//14  1 0
        sum+=fact;//0=0+120=120   120+24=144  120+24+1=145
    }

    if(sum==temp)
      printf("strong");
    else
      printf("not");
       return 0;
}
output:
145
strong
100
not

#include<stdio.h>
long long int factorial(int num)
{

    if(num==0 || num==1)
        return 1;
    return num*factorial(num-1);
}
int main()
{
    int num;
    scanf("%d",&num);
    printf("%lld",factorial(num));
}
